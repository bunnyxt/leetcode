```python
from collections import defaultdict

class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        if len(magazine) < len(ransomNote):
            return False
        
        dr = defaultdict(int)
        for c in ransomNote:
            dr[c] += 1
        
        dm = defaultdict(int)
        for c in magazine:
            dm[c] += 1
            
        result = True
        for char, count in dr.items():
            if dm[char] < count:
                result = False
                break
        
        return result
```
